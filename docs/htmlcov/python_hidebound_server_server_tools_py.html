42def parse_json_file_content(raw_content): 

43 # type: (bytes) -> Dict 

44 ''' 

45 Parses JSON file content as supplied by HTML request. 

46 

47 Args: 

48 raw_content (bytes): Raw JSON file content. 

49 

50 Raises: 

51 ValueError: If header is invalid. 

52 JSONDecodeError: If JSON is invalid. 

53 

54 Returns: 

55 dict: JSON content or reponse dict with error. 

56 ''' 

57 header, content = raw_content.split(',') # type: ignore 

58 temp = header.split('/')[-1].split(';')[0] # type: ignore 

59 if temp != 'json': 

60 msg = f'File header is not JSON. Header: {header}.' # type: ignore 

61 raise ValueError(msg) 

62 

63 output = base64.b64decode(content).decode('utf-8') 

64 return json.loads(output)

67def error_to_response(error): 

68 # type: (Exception) -> flask.Response 

69 ''' 

70 Convenience function for formatting a given exception as a Flask Response. 

71 

72 Args: 

73 error (Exception): Error to be formatted. 

74 

75 Returns: 

76 flask.Response: Flask response. 

77 ''' 

78 args = [' ' + str(x) for x in error.args] # type: Any 

79 args = '\n'.join(args) 

80 klass = error.__class__.__name__ 

81 msg = f'{klass}(\n{args}\n)' 

82 return flask.Response( 

83 response=json.dumps(dict( 

84 error=error.__class__.__name__, 

85 args=list(map(str, error.args)), 

86 message=msg, 

87 code=500, 

88 traceback=traceback.format_exc(), 

89 )), 

90 mimetype='application/json', 

91 status=500,

96def setup_hidebound_directory(root, config_path=None): 

97 # type: (Union[str, Path], Union[str, Path, None]) -> Tuple[Dict, str] 

98 ''' 

99 Creates [root]/hidebound and [root]/hidebound/specifications 

100 directories. Writes a default hidebound config to 

101 [root]/hidebound/hidebound_config.json if one does not exist. 

102 

103 Args: 

104 root (str or Path): Root directory of hidebound data. 

105 config_path (str or Path, optional): Filepath of config data to be 

106 written to [root]/hidebound/hideebiund_config.json. Default: None. 

107 

108 Return: 

109 tuple[dict, str]: Config data and filepath. 

110 ''' 

111 root = Path(root) 

112 hb_root = Path(root, 'hidebound') 

113 os.makedirs(hb_root, exist_ok=True) 

114 os.makedirs(Path(hb_root, 'specifications'), exist_ok=True) 

115 

116 config = { 

117 'root_directory': Path(root, 'projects').as_posix(), 

118 'hidebound_directory': hb_root.as_posix(), 

119 'specification_files': [], 

120 'include_regex': '', 

121 'exclude_regex': r'\.DS_Store', 

122 'write_mode': 'copy' 

123 } 

124 if config_path is not None: 

125 with open(config_path) as f: 

126 config = json.load(f) 

127 

128 config_path = Path(root, 'hidebound', 'hidebound_config.json') 

129 if not config_path.is_file(): 

130 with open(config_path, 'w') as f: 

131 json.dump(config, f, indent=4, sort_keys=True) 

132 

133 return config, config_path.as_posix()
