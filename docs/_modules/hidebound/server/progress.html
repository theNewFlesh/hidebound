[docs]@APP.route('/progress', methods=['GET'])
def create():
    # type: () -> flask.Response
    '''
    Get hidebound app progress.

    Returns:
        Response: Flask Response instance.
    '''
    filepath = Path('/tmp/hidebound/logs/progress/hidebound-progress.log')
    state = {}
    if filepath.is_file():
        state = ProgressLogger.read(filepath)[-1]
    return flask.Response(
        response=json.dumps(state),
        mimetype='application/json'
    )
